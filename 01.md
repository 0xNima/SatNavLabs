## b)

**From reference ellipsoid to Geocentric reference frame:**

$$
x = (N + h)\cos(\varphi)\cos(\lambda)
$$

$$
y = (N + h)\cos(\varphi)\sin(\lambda)
$$

$$
z = \left((1 - e^2)N + h\right)\sin(\varphi)
$$

$$
N = \frac{a}{\sqrt{1 - e^2 \sin^2(\varphi)}}
$$

$$
e^2 = 2f - f^2
$$

```matlab
clc; clear all;

a = 6378137;
D = 298.257223563;
f = 1/D;

format long;

%% part-b
function geocen = detic2centric(lat, long, h, a, f)
    e2 = 2*f - f^2;
    N = a / sqrt(1-e2*sind(lat)^2);
    geocen = [
        (N + h) * cosd(lat) * cosd(long);
        (N + h) * cosd(lat) * sind(long);
        ((1 - e2) * N + h) * sind(lat)
    ];
end

A = [dms2degrees([48 46 50]) dms2degrees([9 10 17]) 320];
B = [dms2degrees([52 30 58.540]) dms2degrees([13 22 39.767]) 71.371];

accurateA = detic2centric(A(1), A(2), A(3), a, f);
accurateB = detic2centric(B(1), B(2), B(3), a, f);

fprintf("\nf=%.10f -----------------------------------------\n", f)
fprintf("A: (%f, %f, %f) \n", accurateA);
fprintf("B: (%f, %f, %f) \n", accurateB);

%% A: (4157199.017754, 671189.101833, 4774753.823636) 
%% B: (3784037.484006, 899929.576054, 5038022.562017)
```

```matlab
%% find the decimal point (trial and error)
dp = 2;
e = 0.001; % 1 mm
while 1
    rounded_denom = round(D, dp);
    rounded_f = 1/rounded_denom;
    unaccurateA = detic2centric(A(1), A(2), A(3), a, rounded_f);
    dA = abs(accurateA - unaccurateA);
    dPosition = sqrt(sum(dA .^ 2));
    if dPosition < e
        break
    end
    dp = dp + 1;
end

fprintf("\nDecimal points: %d\n", dp);
```
