> ... the message structure shall utilize a basic format of a 1500 bit long frame made up of
five subframes, each subframe being 300 bits long. Subframes 4 and 5 shall be subcommutated 25 times each,

$\frac{1500}{50 \mathrm{Hz}} = 30 \ \mathrm{s} $
#### Complete data message:
$\frac{25 \times 30 }{60} = 12.5 \mathrm{minutes} $

#### 1 frame: 1500 bits $\implies$ 5 subframes: 300 bits each

## b)
```python
from pathlib import Path

SUB_FRAMES = 5
SUB_FRAMES_BIT_LENGTH = 300
TOTAL_BITS = 1500

# pre-allocation
subframes = [
    [0] * SUB_FRAMES_BIT_LENGTH for _ in range(SUB_FRAMES)
]

subframe_ids = [0] * SUB_FRAMES


def bits_to_int(bits):
    val = 0
    for b in bits:
        val = (val << 1) | b
    return val


with open (Path(__file__).parent / 'NavMessage_ss26.txt', 'r') as file:
    for i in range(TOTAL_BITS):
        bit = int(file.readline().strip())
        current_subframe = i // SUB_FRAMES_BIT_LENGTH
        current_subframe_index = i % SUB_FRAMES_BIT_LENGTH

        subframes[current_subframe][current_subframe_index] = bit

        if current_subframe_index == 52: # 30 (TLM offset) + 22 (HOW offset)
            bits = subframes[current_subframe][49: 52]
            subframe_ids[current_subframe] = bits_to_int(bits)

```

#### Result: $[2, 4, 5, 3, 1]$

# Task 3
```python
from datetime import timedelta

...


ordered_subframes_idx = dict()

for i, sid in enumerate(subframe_ids):
    ordered_subframes_idx[sid] = i


def extract(subframe_idx, scale_factor, offset, limit, twos_complement=False):
    bits = subframes[ordered_subframes_idx[subframe_idx]][offset: offset + limit]
    integer = bits_to_int(bits)

    if twos_complement:
        mask = 1 << (len(bits) - 1)
        if integer & mask:  # MSB = 1
            integer -= (1 << len(bits))

    return integer * scale_factor

# extract some data
# TOE
toe = extract(subframe_idx=2, scale_factor=2**4, offset=270, limit=16)
week_day, seconds = divmod(toe, 86400)

GPS_DAYS = [
    "Sunday", "Monday", "Tuesday", "Wednesday", "Thursday", "Friday", "Saturday"
]

print(f'TOE: {GPS_DAYS[week_day]} {timedelta(seconds=seconds)}')
```
